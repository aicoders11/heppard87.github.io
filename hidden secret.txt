        // Utility Functions
        function initAudio() {
            if (gameState.audioCtx || gameState.isMuted) return;
            try {
                gameState.audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                startHeartbeat();
            } catch(e) {
                console.log("Audio not supported");
            }
        }

        function startHeartbeat() {
            if (gameState.heartbeat) clearInterval(gameState.heartbeat);
            if (gameState.isMuted) return;
            
            const speed = Math.max(300, gameState.dist * 100);
            
            gameState.heartbeat = setInterval(() => {
                if (gameState.isMuted) return;
                const osc = gameState.audioCtx.createOscillator();
                const gain = gameState.audioCtx.createGain();
                osc.frequency.setValueAtTime(50, gameState.audioCtx.currentTime);
                gain.gain.setValueAtTime(0.2, gameState.audioCtx.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.01, gameState.audioCtx.currentTime + 0.1);
                osc.connect(gain); 
                gain.connect(gameState.audioCtx.destination);
                osc.start(); 
                osc.stop(gameState.audioCtx.currentTime + 0.1);
            }, speed);
        }

        function creepySpeak(text) {
            if (gameState.isMuted) return;
            const msg = new SpeechSynthesisUtterance(text);
            msg.pitch = 0.1;
            msg.rate = 0.4;
            msg.volume = 0.5;
            window.speechSynthesis.speak(msg);
        }

        function triggerScare() {
            const overlay = document.getElementById('scare-overlay');
            overlay.style.opacity = '1';
            overlay.style.pointerEvents = 'auto';
            document.body.classList.add('shaking');
            document.getElementById('game-window').style.filter = "blur(2px)";
            
            creepySpeak("I FOUND YOU");
            
            setTimeout(() => {
                overlay.style.opacity = '0';
                overlay.style.pointerEvents = 'none';
                document.body.classList.remove('shaking');
                document.getElementById('game-window').style.filter = "none";
                gameState.sanity -= 20;
                gameState.dist = 3;
                updateUI();
            }, 1500);
        }

        function triggerShake() {
            document.body.classList.add('shaking');
            setTimeout(() => document.body.classList.remove('shaking'), 500);
        }

        function triggerBloodDrips() {
            const container = document.getElementById('blood-effects');
            for (let i = 0; i < 3; i++) {
                setTimeout(() => {
                    const drip = document.createElement('div');
                    drip.className = 'blood-drip active';
                    drip.style.left = Math.random() * 100 + '%';
                    container.appendChild(drip);
                    setTimeout(() => drip.remove(), 4000);
                }, i * 300);
            }
        }

        function showNotification(title, msg) {
            const toast = document.getElementById('toast');
            document.getElementById('toast-title').textContent = title;
            document.getElementById('toast-msg').textContent = msg;
            toast.classList.add('visible');
            setTimeout(() => toast.classList.remove('visible'), 2500);
        }

        function updateUI() {
            const room = rooms[gameState.location];
            
            // Update stats
            document.getElementById('sanity-val').textContent = gameState.sanity;
            document.getElementById('dist-val').textContent = gameState.dist;
            document.getElementById('progress-val').textContent = gameState.progress;
            document.getElementById('inv-val').textContent = gameState.inventory.join(', ') || 'Empty';
            document.getElementById('progress-bar').style.width = gameState.progress + '%';
            
            // Critical sanity styling
            const sanityEl = document.getElementById('sanity-val');
            if (gameState.sanity < 30) {
                sanityEl.classList.add('critical');
                document.body.style.filter = "contrast(1.3) grayscale(0.4) sepia(0.3)";
                if (Math.random() > 0.7) triggerBloodDrips();
            } else if (gameState.sanity < 60) {
                sanityEl.classList.remove('critical');
                document.body.style.filter = "contrast(1.1) brightness(0.9)";
            } else {
                sanityEl.classList.remove('critical');
                document.body.style.filter = "none";
            }

            // Update content
            document.getElementById('title').innerText = room.title;
            document.getElementById('story-text').innerHTML = room.text;
            
            // Build choices
            const choicesDiv = document.getElementById('choices');
            choicesDiv.innerHTML = '';
            
            if (room.ending) {
                const btn = document.createElement('button');
                btn.innerText = "↻ PLAY AGAIN";
                btn.style.background = "rgba(139, 0, 0, 0.4)";
                btn.onclick = () => location.reload();
                choicesDiv.appendChild(btn);
                return;
            }
            
            room.actions.forEach(a => {
                if (a.require && !gameState.inventory.includes(a.require)) return;
                
                const btn = document.createElement('button');
                btn.innerHTML = a.text + (a.sanity ? ` <span style="color:#8b0000;float:right;">[Sanity ${a.sanity}]</span>` : '');
                btn.onclick = () => takeAction(a);
                choicesDiv.appendChild(btn);
            });
        }

        function takeAction(a) {
            initAudio();
            
            if (a.move) {
                gameState.dist = Math.max(1, gameState.dist - 1);
                if (gameState.dist <= 2 && !gameState.isMuted) creepySpeak("So close");
            }
            if (a.sanity) gameState.sanity = Math.max(0, Math.min(100, gameState.sanity + a.sanity));
            if (a.item && !gameState.inventory.includes(a.item)) {
                gameState.inventory.push(a.item);
                showNotification("Acquired", a.item);
            }
            if (a.target) gameState.location = a.target;
            if (a.flags) Object.assign(gameState.flags, a.flags);
            
            const room = rooms[gameState.location];
            if (!gameState.visited.has(gameState.location)) {
                gameState.visited.add(gameState.location);
                gameState.progress = Math.max(gameState.progress, room.progress || 0);
            }
            
            // Special effects
            if (room.instant === 'scare') triggerScare();
            if (room.instant === 'shake') triggerShake();
            if (room.sanity) gameState.sanity = Math.max(0, Math.min(100, gameState.sanity + room.sanity));
            
            // Random whisper
            if (Math.random() > 0.85 && gameState.sanity < 50) {
                const whispers = ["Behind you", "Don't look", "Run", "Join us"];
                creepySpeak(whispers[Math.floor(Math.random() * whispers.length)]);
            }
            
            // Blood drips on low sanity
            if (gameState.sanity < 40 && Math.random() > 0.6) triggerBloodDrips();
            
            // Stalker catch
            if (gameState.dist <= 0 || gameState.sanity <= 0) {
                gameState.location = 'death_stalker';
            }
            
            startHeartbeat();
            updateUI();
        }

        function saveGame() {
            const saveData = {
                sanity: gameState.sanity,
                location: gameState.location,
                inventory: gameState.inventory,
                dist: gameState.dist,
                progress: gameState.progress,
                flags: gameState.flags,
                visited: Array.from(gameState.visited)
            };
            localStorage.setItem('hiddenSecretsSave', JSON.stringify(saveData));
            showNotification("Game Saved", "Progress stored in blood");
        }

        function loadGame() {
            const saved = localStorage.getItem('hiddenSecretsSave');
            if (saved) {
                const data = JSON.parse(saved);
                gameState.sanity = data.sanity;
                gameState.location = data.location;
                gameState.inventory = data.inventory;
                gameState.dist = data.dist;
                gameState.progress = data.progress;
                gameState.flags = data.flags || {};
                gameState.visited = new Set(data.visited || []);
                
                updateUI();
                showNotification("Game Loaded", "Nightmare resumed");
                initAudio();
            } else {
                showNotification("No Save", "Nothing to load");
            }
        }

        function toggleMute() {
            gameState.isMuted = !gameState.isMuted;
            document.getElementById('mute-btn').textContent = gameState.isMuted ? '🔇' : '🔊';
            if (gameState.isMuted) {
                if (gameState.heartbeat) clearInterval(gameState.heartbeat);
            } else {
                initAudio();
            }
        }

        // Initialize
        window.onload = () => {
            updateUI();
            
            // Initial blood drip
            setTimeout(triggerBloodDrips, 2000);
        };
